20 Templated vector class much like stl vector container
21
22 NOTE: elem[n] is vector component n for all n >= 0 AND n < size_v
23 size_v = the number of items stored in the vector

25 if size_v < space there is space for (space - size_v) elements after

147 void push_back(T newObj)
148 // increase vector size by one; initialize the new element with newObj
149 {
150 if (space == 0)
151 reserve(8); // start with space for 8 elements
152 else if (size_v == space)
153 reserve(2 * space); // get more space
154 // add newObj at end
155 elem[size_v] = newObj;

160 void reserve(int newalloc)
161 {
162 // never decrease allocation
163 if (newalloc > space)
164 {
165 // allocate new space
166 T *p = new T[newalloc];
167 // copy old elements (do a move)
168 for (int i = 0; i < size_v; ++i)
169 p[i] = std::move(elem[i]);
170 // deallocate old space
171 delete[] elem;
172 // reset elem
173 elem = p;
174 // reset vector capacity
175 space = newalloc;
176 }
177 }

210 // insert a new element val before p
211 iterator insert(iterator p, const T& newObj)
212 {
213 // check that iterator is in bounds
214 if (p >= begin() && p < end())
215 {
216 // where to insert (index is distance between p and begin())
217 // (need to calculate this offset because p will become an invalid
218 // iterator if reserve is called, as new memory - meaning new
219 // location in meory - will be allocated)
220 int index = p - begin();
221
222 // make sure we have space
223 if (size_v + 1 > space)
224 reserve(2 * space);
225
226 // copy elements that are after the desired index one position to
227 // the right to make room for the new element
228 for (iterator it = end(); it > (begin() + index); --it)
229 *it = *(it - 1);
230
231 // insert value
232 *(elem + index) = newObj;
233 // increment vector's size
234 ++size_v;
235 // return iterator pointing to newly inserted element
236 return elem + index;
237 }
238 // return nullptr if p was out of bounds
239 return nullptr;
240 }
241
242 iterator erase(iterator p) // remove element pointed to by p
243 {
244 // check bounds of p (to avoid seg fault)
245 if (p < begin() || p >= end())
246 return nullptr;
247 for (iterator pos = p + 1; pos != end(); ++pos)
248 *(pos - 1) = *pos; // copy element one position to the left
249 // delete (end() - 1);
250 --size_v;
251 return p;
252 }
