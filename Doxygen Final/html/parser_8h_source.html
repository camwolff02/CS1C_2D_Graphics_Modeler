81myStd::vector<Shape*> parse(QString filename = "shapes.txt")
82{
83 // create vector to write to
84 myStd::vector<Shape*> vect;
85
86 // fix file path before opening
87 QString fullPath = QDir::currentPath() + '/' + filename;
88 fullPath = fullPath.replace('/', "\\\\");
89
90 // open file using Qt's file system
91 QFile file {filename};
92
93 // give error if we can't open the file in read only
94 // QIODevice::Text turns Windows style to c++ style line terminators
95 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
96 QMessageBox::information(0, "error", file.errorString());
97 return vect;
98 }
99 else {
100 qInfo() << "file read successfully\n";
101 }
102
103 // open Qt text stream to write file to
104 QTextStream in {&file};
105 while (!in.atEnd()) { // while we are not at file end
106 // create new shape to add to vector
107 Shape* shape = nullptr;
108 in.readLine();
109
110 // first 3 param are same for all shapes
111 int id = in.readLine().remove(0, 9).toInt();
112 QString type = in.readLine().remove(0, 11);
113 QList<int> dims;
114 foreach (QString str, in.readLine().remove(0, 17).split(", ")) dims.append(str.toInt());
115
116 // now handling rest of param
117 if (type == "Text") {
118 QString msg = in.readLine().remove(0, 12);
119 QColor color = in.readLine().remove(0, 11);
120 QString alignment = in.readLine().remove(0, 15);
121 int pointSize = in.readLine().remove(0, 15).toInt();
122 QString fontFamily = in.readLine().remove(0, 16);
123 QString fontStyle = in.readLine().remove(0, 15);
124 QString fontWeight = in.readLine().remove(0, 16);
125
126 shape = new myStd::Text(msg, makeFont(fontFamily, pointSize, fontWeight, fontStyle),
127 color, dims.at(0), dims.at(1), dims.at(2), dims.at(3), id);
128 }
129 else {
130 QColor color = in.readLine().remove(0, 10);
131 int width = in.readLine().remove(0, 10).toInt();
132 Qt::PenStyle style = getPenStyle(in.readLine().remove(0, 10));
133 Qt::PenCapStyle capStyle = getPenCapStyle(in.readLine().remove(0, 13));
134 Qt::PenJoinStyle joinStyle = getPenJoinStyle(in.readLine().remove(0, 14));
135
136 if (type != "Line" && type != "Polyline") {
137 QColor brushColor = in.readLine().remove(0, 12);
138 QString brushStyle = in.readLine().remove(0, 12);
139
140 if (type == "Polygon") {
141 shape = new myStd::Polygon(id);
142 for (int i = 0; i < dims.length(); i+=2)
143 ((myStd::Polygon*)shape)->addPoint(dims.at(i), dims.at(i));
144 }
145 else if (type == "Rectangle") {
146 shape = new myStd::Rectangle(id, dims.at(0), dims.at(1), dims.at(2), dims.at(3));
147 }
148 else if (type == "Square") {
149 shape = new myStd::Rectangle(id, dims.at(0), dims.at(1), dims.at(2), dims.at(2));
150 }
151 else if (type == "Ellipse") {
152 shape = new myStd::Ellipse(id, dims.at(2), dims.at(3));
153 shape->move(dims.at(0), dims.at(1));
154 }
155 else {
156 shape = new myStd::Ellipse(id, dims.at(2), dims.at(2));
157 shape->move(dims.at(0), dims.at(1));
158 }
159
160 shape->setBrush(brushColor, getBrushStyle(brushStyle));
161 }
162 else {
163 if (type == "Line") {
164 shape = new myStd::Line(id, dims.at(0), dims.at(1), dims.at(2), dims.at(3));
165 }
166 else { // type == "Polyline"
167 shape = new myStd::Polyline(id);
168 for (int i = 0; i < dims.length(); i+=2)
169 ((myStd::Polyline*)shape)->addPoint(dims.at(i), dims.at(i));
170 }
171 }
172 shape->setPen(color, width, style, capStyle, joinStyle);
173 }
174
175 if (shape != nullptr) vect.push_back(shape);
176 }
177 qInfo("\nvector write complete\n");
178 file.close();
179 return vect;
180}
